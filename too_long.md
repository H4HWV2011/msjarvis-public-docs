# -----------------------------------------------------------------------
# DOWNLOAD
# -----------------------------------------------------------------------
if [[ "$MODE" == "download" || "$MODE" == "both" ]]; then
  for ZIP in "${ZIPS[@]}"; do
    URL="${BASE_URL}/${ZIP}"
    DEST="${GIS_DIR}/$(basename "$ZIP")"
    if [[ -f "$DEST" ]]; then
      echo "[SKIP] $(basename "$ZIP") already exists" | tee -a "$LOG"
      continue
    fi
    echo "[DL] $URL" | tee -a "$LOG"
    wget -q --show-progress -O "$DEST" "$URL" 2>&1 | tee -a "$LOG"
    if [[ ${PIPESTATUS[0]} -ne 0 ]]; then
      echo "[FAIL] $URL" | tee -a "$LOG"
      rm -f "$DEST"
    fi
  done
fi

# -----------------------------------------------------------------------
# IMPORT
# -----------------------------------------------------------------------
if [[ "$MODE" == "import" || "$MODE" == "both" ]]; then
  for ZIP in "${GIS_DIR}"/*.zip; do
    BASENAME=$(basename "$ZIP" .zip)
    EXTRACT_DIR="${GIS_DIR}/extracted/${BASENAME}"
    mkdir -p "$EXTRACT_DIR"
    unzip -q -o "$ZIP" -d "$EXTRACT_DIR"

    # Sanitize table name
    TABLE=$(echo "$BASENAME" | tr '[:upper:]' '[:lower:]' | sed 's/[^a-z0-9]/_/g' | cut -c1-63)

    SHP=$(find "$EXTRACT_DIR" -name "*.shp" | head -1)
    if [[ -z "$SHP" ]]; then
      echo "[SKIP] No SHP in $BASENAME" | tee -a "$LOG"
      continue
    fi

    echo "[IMPORT] $TABLE <- $SHP" | tee -a "$LOG"
    ogr2ogr \\
      -f "PostgreSQL" \\
      "PG:host=localhost port=${PG_PORT} dbname=${DB} user=${DB_USER}" \\
      "$SHP" \\
      -nln "$TABLE" \\
      -nlt PROMOTE_TO_MULTI \\
      -lco GEOMETRY_NAME=geom \\
      -lco FID=ogc_fid \\
      -t_srs EPSG:4326 \\
      -overwrite \\
      -progress 2>&1 | tee -a "$LOG"
  done
fi

echo "=== Done $(date) ===" | tee -a "$LOG"
